## Executive Summary

### Project Goals
- Create a browser-based photo glitch effects editor
- Leverage Three.js for GPU-accelerated image processing
- Build a modular, extensible architecture for adding new effects
- Deliver responsive UI that works on desktop and mobile
- All processing client-side (no server required)

### Key Features (MVP)
- Photo upload from file system
- Camera capture (WebRTC)
- 2 initial effects: RGB Shift + Pixel Sort
- Real-time preview with adjustable parameters
- Export processed image (PNG/JPEG)

### C. Full-Screen Quad Setup

```typescript
// src/components/canvas/ImagePlane.tsx

function ImagePlane({ texture }: { texture: THREE.Texture }) {
  const { viewport } = useThree();
  
  // Calculate aspect-correct sizing
  const imageAspect = texture.image.width / texture.image.height;
  const viewportAspect = viewport.width / viewport.height;
  
  let scale: [number, number, number];
  if (imageAspect > viewportAspect) {
    // Image is wider - fit to width
    scale = [viewport.width, viewport.width / imageAspect, 1];
  } else {
    // Image is taller - fit to height
    scale = [viewport.height * imageAspect, viewport.height, 1];
  }
  
  return (
    <mesh scale={scale}>
      <planeGeometry args={[1, 1]} />
      <effectMaterial texture={texture} />
    </mesh>
  );
}
```

// src/hooks/useCanvasExport.ts

function useCanvasExport() {
  const exportImage = async (
    format: 'png' | 'jpeg',
    quality: number = 0.9
  ) => {
    // Create off-screen renderer at full resolution
    const offscreenRenderer = new THREE.WebGLRenderer({
      preserveDrawingBuffer: true,
      antialias: true,
    });
    
    const { width, height } = store.getState().dimensions;
    offscreenRenderer.setSize(width, height);
    
    // Render scene
    offscreenRenderer.render(scene, camera);
    
    // Extract as data URL
    const mimeType = format === 'png' ? 'image/png' : 'image/jpeg';
    const dataUrl = offscreenRenderer.domElement.toDataURL(mimeType, quality);
    
    // Trigger download
    const link = document.createElement('a');
    link.download = `glitch-${Date.now()}.${format}`;
    link.href = dataUrl;
    link.click();
    
    // Cleanup
    offscreenRenderer.dispose();
  };
  
  return { exportImage };
}
